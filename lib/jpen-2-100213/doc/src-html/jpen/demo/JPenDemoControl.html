049            final PenCanvas penCanvas;
050            final MainPanel mainPanel;
051            final JButton statusReportButton=new JButton("Status Report...");
052    
053            public JPenDemoControl(){
054                    penCanvas=new PenCanvas();
055                    mainPanel=new MainPanel(penCanvas);
056    
057                    setSupportCustomPKinds(true);
058    
059                    statusReportButton.addActionListener(new ActionListener(){
060                                            //@Override
061                                            public void actionPerformed(ActionEvent ev){
062                                                    StatusReportPanel statusReportPanel=new StatusReportPanel(
063                                                                            new StatusReport(penCanvas.penManager));
064                                                    //statusReportPanel.panel.setPreferredSize(SIZE);
065                                                    JOptionPane.showMessageDialog(mainPanel.panel, statusReportPanel.panel, "JPen Status Report", JOptionPane.INFORMATION_MESSAGE);
066                                            }
067                                    });
068    
069                    /*JFrame f=new JFrame("JPen Demo");
070                    f.getContentPane().add(mainPanel.panel);
071                    f.setSize(new Dimension(450, 480));
072                    f.setResizable(false);
073                    f.setVisible(true);
074                    f.setDefaultCloseOperation(f.EXIT_ON_CLOSE);*/
075            }
076    
077            public void setSupportCustomPKinds(boolean supportCustomPKinds){
078                    mainPanel.devicesPanel.devicesTableModel.
079                    setSupportCustomPKinds(supportCustomPKinds);
080            }
081    
082            public JComponent getMainPanelComponent(){
083                    return mainPanel.panel;
084            }
085    
086            public JButton getStatusReportButton(){
087                    return statusReportButton;
088            }
089    
090            public void showDialog(JComponent parent, String title){
091                    String closeOption="Close";
092                    Object[] options=new Object[]{statusReportButton, closeOption};
093                    JOptionPane.showOptionDialog(null, getMainPanelComponent(), title,
094                                    JOptionPane.DEFAULT_OPTION, JOptionPane.PLAIN_MESSAGE,
095                                    null, options, closeOption);
096            }
097    
098            public static void main(String... args) throws IOException, NumberFormatException{
099                    setupLookAndFeel();
100                    JPenDemoControl jpenDemoControl=new JPenDemoControl();
101                    startInspector(jpenDemoControl.penCanvas.penManager);
102                    jpenDemoControl.showDialog(null, "JPen Demo");
103                    System.exit(0);
104            }
105    
106            static void setupLookAndFeel(){
107                    try{
108                            if(System.getProperty("os.name").toLowerCase().contains("linux"))
109                                    UIManager.setLookAndFeel("com.sun.java.swing.plaf.gtk.GTKLookAndFeel");
110                            else
111                                    UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
112                    }catch(Exception ex){
113                            L.warning("The \"system\" look and feel couldn't be set.");
114                    }
115            }
116    
117            static void startInspector(PenManager penManager) throws IOException{
118                    String inspectorPeriodProperty=System.getProperty("jpen.demo.inspectorPeriod");
119                    if(inspectorPeriodProperty!=null){
120                            int inspectorPeriod=Integer.valueOf(inspectorPeriodProperty);
121                            Inspector inspector=new Inspector(penManager, "jpen", inspectorPeriod);
122                            L.info("inspector constructed");
123                    }
124            }
