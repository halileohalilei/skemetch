034    public class Pen extends PenState {
035            private static final Logger L=Logger.getLogger(Pen.class.getName());
036            //static{L.setLevel(Level.ALL);}
037    
038            public static final int DEFAULT_FREQUENCY=60; // TODO: 50 is a better default or less??
039    
040            public final PenManager penManager;
041            private int frequency;
042            private volatile MyThread thread;
043    
044            /** Tail of event queue. */
045            private PenEvent lastDispatchedEvent=new PenEvent.Dummy();
046            final PenScheduler scheduler;
047            public final PenState lastScheduledState;
048            private final List<PenListener> listeners=new ArrayList<PenListener>();
049            private PenListener[] listenersArray;
050            private boolean firePenTockOnSwing;
051            public final PLevelEmulator levelEmulator;
052    
053            private final class MyThread
054                    extends Thread {
055                    final int periodMillis;
056                    long beforeTime;
057                    long waitTime;
058                    long availablePeriod;
059                    PenEvent event;
060                    boolean waitedNewEvents;
061                    Exception exception;
062                    private final Waiter waiter=new Waiter();
063                    volatile boolean stopRunning;
064                    Thread oldThread;
065    
066                    final class Waiter
067                            extends Object{
068                            synchronized void doWait(long timeout) throws InterruptedException{
069                                    if(!stopRunning)
070                                            wait(timeout);
071                            }
072                            synchronized void doNotify(){
073                                    notify();
074                            }
075                    }
076    
077                    MyThread(Thread oldThread){
078                            periodMillis=1000/Pen.this.frequency;
079                            this.oldThread=oldThread;
080                            setName("jpen-Pen-["+periodMillis+"ms]");
081                            setDaemon(true);
082                    }
083                    private final Runnable penTockFirer=new Runnable(){
084                                            //@Override
085                                            public void run(){
086                                                    //System.out.println("firing tocks "+System.currentTimeMillis());
087                                                    for(PenListener l:getListenersArray())
088                                                            l.penTock(availablePeriodLeft());
089                                            }
090                                    };
091    
092                    public void run() {
093                            try {
094                                    L.finest("v");
095                                    if(oldThread!=null)
096                                            oldThread.join();
097                                    oldThread=null;
098                                    while(!stopRunning) {
099                                            waitedNewEvents=waitNewEvents();
100                                            beforeTime=System.currentTimeMillis();
101                                            if(waitedNewEvents)
102                                                    waitTime=0;
103                                            else
104                                                    yield();
105                                            while((event=lastDispatchedEvent.next)!=null && event.getTime()<=beforeTime) {
106                                                    event.copyTo(Pen.this);
107                                                    event.dispatch();
108                                                    lastDispatchedEvent.next=null;
109                                                    lastDispatchedEvent=event;
110                                            }
111                                            availablePeriod=periodMillis+waitTime; // waitTime here is always <=0, if it is <0 then the whole processing of the previous round took longer than the time available. 
112                                            //System.out.println("going to fire tock "+System.currentTimeMillis());
113                                            firePenTock();
114                                            waitTime=availablePeriodLeft();// the same:  (periodMillis-evalCurrentProcTime())+waitTime;
115                                            if(waitTime>0) {
116                                                    //System.out.println("going to wait: "+waitTime);
117                                                    ThreadUtils.sleepUninterrupted(waitTime);
118                                                    //waiter.doWait(waitTime); // in some cases, this (instead of sleepUninterrupted ^ ) gives better overall performance (wintab-pulling, jpen demo). We can put this as an alternate behavior if  needed.
119                                                    waitTime=0;
120                                            }
121                                    }
122                            } catch(Exception ex) {
123                                    L.severe("jpen-Pen thread threw an exception: "+ThrowableUtils.evalStackTraceString(ex));
124                                    exception=ex;
125                            }
126                            L.finest("^");
127                    }
128    
129                    private long evalCurrentProcTime(){
130                            return System.currentTimeMillis()-beforeTime;
131                    }
132                    
133                    private long availablePeriodLeft(){
134                            return availablePeriod-evalCurrentProcTime();
135                    }
136    
137                    private boolean waitNewEvents() throws InterruptedException {
138                            if(lastDispatchedEvent.next!=null)
139                                    return false;
140                            waiter.doWait(0);
141                            return true;
142                    }
143    
144                    private void firePenTock() throws InterruptedException, InvocationTargetException{
145                            if(getListenersArray().length==0)
146                                    return;
147                            if(firePenTockOnSwing)
148                                    SwingUtilities.invokeAndWait(penTockFirer);
149                            else
150                                    penTockFirer.run();
151                    }
152    
153                    void processNewEvents() {
154                            waiter.doNotify();
155                    }
156    
157                    void stop(boolean join){
158                            stopRunning=true;
159                            processNewEvents(); // because it may be waiting for new events.
160                            if(join)
161                                    try{
162                                            join();
163                                    }
164                                    catch(InterruptedException ex) {
165                                            throw new Error(ex);
166                                    }
167                    }
168            }
169    
170            Pen(PenManager penManager) {
171                    this.penManager=penManager;
172                    this.scheduler=new PenScheduler(this);
173                    this.lastScheduledState=scheduler.lastScheduledState;
174                    this.levelEmulator=new PLevelEmulator(this);
175                    setFrequencyLater(DEFAULT_FREQUENCY);
176            }
177    
178            void processNewEvents(){
179                    thread.processNewEvents();
180            }
181    
182            PenEvent getLastDispatchedEvent(){
183                    return lastDispatchedEvent;
184            }
185    
186            public boolean getFirePenTockOnSwing() {
187                    return firePenTockOnSwing;
188            }
189    
190            /**
191            @param firePenTockOnSwing If {@code true} then {@link PenListener#penTock(long)} is called from the event dispatch thread. {@code false} by default.
192            */
193            public void setFirePenTockOnSwing(boolean firePenTockOnSwing){
194                    this.firePenTockOnSwing = firePenTockOnSwing;
195            }
196    
197            /**
198            Changes the event firing frequency. The pen collects device (tablet) data points and stores them in a buffer. The data  points are taken from this buffer and fired as {@link PenEvent}s at this frequency.<p> 
199            
200            This method returns immediately, the change of frequency will happen after all the pending events are processed.
201            
202            @see #addListener(PenListener) 
203            @see #removeListener(PenListener)
204            */
205            public void setFrequencyLater(int frequency){
206                    setFrequency(frequency, false);
207            }
208            
209                    private synchronized void setFrequency(int frequency, boolean wait) {
210                    if(frequency<=0)
211                            throw new IllegalArgumentException();
212                    if(frequency==this.frequency)
213                            return;
214                    if(wait && SwingUtilities.isEventDispatchThread())
215                            throw new Error("Cannot call setFrequency(int, <true>) from the event dispatcher thread");
216                    L.finest("v");
217                    MyThread oldThread=this.thread;
218                    if(oldThread!=null){
219                            oldThread.stop(wait);
220                    }
221                    this.frequency=frequency;
222                    thread=new MyThread(oldThread);
223                    thread.start();
224                    L.finest("^");
225            }
226    
227            public int getFrequency() {
228                    return frequency;
229            }
230    
231            public int getPeriodMillis(){
232                    return thread.periodMillis;
233            }
234    
235            public synchronized Exception getThreadException(){
236                    return thread.exception;
237            }
238    
239            /**
240            Adds a {@link PenListener} for {@link PenEvent}s fired by this pen.
241            */
242            public void addListener(PenListener l) {
243                    synchronized(listeners) {
244                            listeners.add(l);
245                            listenersArray=null;
246                    }
247            }
248    
249            /**
250            Removes a {@link PenListener} previously added using {@link #addListener(PenListener)}.
251            */
252            public void removeListener(PenListener l) {
253                    synchronized(listeners) {
254                            listeners.remove(l);
255                            listenersArray=null;
256                    }
257            }
258    
259            PenListener[] getListenersArray() {
260                    synchronized(listeners){
261                            if(listenersArray==null)
262                                    listenersArray=listeners.toArray(new PenListener[listeners.size()]);
263                            return listenersArray;
264                    }
265            }
