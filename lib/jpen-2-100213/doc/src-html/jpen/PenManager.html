024    import java.util.Collections;

037    public final class PenManager {
038            private static final Logger L=Logger.getLogger(PenManager.class.getName());
039            
040            public static String getJPenFullVersion(){
041                    return BuildInfo.getFullVersion();
042            }
043    
044            public final Pen  pen=new Pen(this);
045            public final PenOwner penOwner;
046            private final Set<PenProvider.Constructor> providerConstructors=new HashSet<PenProvider.Constructor>();
047            private final Set<PenProvider.Constructor> providerConstructorsA=Collections.unmodifiableSet(providerConstructors);
048            private final Map<Byte, PenDevice> deviceIdToDevice=new HashMap<Byte, PenDevice>(Byte.MAX_VALUE, 1f);
049            private final Collection<PenDevice> devicesA=Collections.unmodifiableCollection(deviceIdToDevice.values());
050            private byte nextDeviceId;
051            private volatile boolean paused=true;
052            private final List<PenManagerListener> listeners=new ArrayList<PenManagerListener>();
053            private PenManagerListener[] listenersArray;
054            final PenDevice emulationDevice;
055            private PenProvider systemMouseProvider; // may be null
056    
057            public PenManager(Component component) {
058                    this(new AwtPenOwner(component));
059            }
060    
061            public PenManager(PenOwner penOwner){
062                    this.penOwner=penOwner;
063                    synchronized(pen.scheduler){
064                            PenProvider.Constructor emulationProviderConstructor=new EmulationProvider.Constructor();
065                            addProvider(emulationProviderConstructor);
066                            @SuppressWarnings("unchecked")
067                            EmulationProvider emulationProvider=(EmulationProvider)emulationProviderConstructor.getConstructed();
068                            emulationDevice=emulationProvider.device;
069    
070                            for(PenProvider.Constructor penProviderConstructor: penOwner.getPenProviderConstructors())
071                                    addProvider(penProviderConstructor);
072                            penOwner.setPenManagerHandle(new PenOwner.PenManagerHandle(){
073                                                    //@Override
074                                                    public final PenManager getPenManager(){
075                                                            return PenManager.this;
076                                                    }
077                                                    //@Override
078                                                    public final Object getPenSchedulerLock(){
079                                                            return pen.scheduler;
080                                                    }
081                                                    //@Override
082                                                    public final void setPenManagerPaused(boolean paused){
083                                                            PenManager.this.setPaused(paused);
084                                                    }
085                                            });
086                    }
087            }
088            
089            /**
090            @return the mouse PenProvider or {@code null} if no mouse provider has been added.
091            @see #addProvider(PenProvider.Constructor)
092            */
093            public PenProvider getSystemMouseProvider(){
094                    return systemMouseProvider;
095            }
096            
097            public boolean isSystemMouseDevice(PenDevice device){
098                    return device.getProvider()==systemMouseProvider;
099            }
100    
101            /**
102            Constructs and adds provider if {@link PenProvider.Constructor#constructable(PenManager)} is true.
103            @return The {@link PenProvider} added or null if it couldn't be constructed.
104            */
105            private  PenProvider addProvider(PenProvider.Constructor providerConstructor) {
106                    if(providerConstructor.constructable(this)) {
107                            if(!this.providerConstructors.add(providerConstructor))
108                                    throw new IllegalArgumentException("constructor already added");
109                            if(providerConstructor.construct(this)){
110                                    PenProvider provider=providerConstructor.getConstructed();
111                                    if(provider instanceof SystemProvider){
112                                            if(this.systemMouseProvider!=null){
113                                                    providerConstructors.remove(providerConstructor);
114                                                    throw new IllegalStateException("system provider already added");
115                                            }
116                                            this.systemMouseProvider=provider;
117                                    }
118                                    for(PenDevice device:provider.getDevices())
119                                            firePenDeviceAdded(providerConstructor, device);
120                                    return provider;
121                            }
122                    }
123                    return null;
124            }
125    
126            public void addListener(PenManagerListener l) {
127                    synchronized(listeners) {
128                            listeners.add(l);
129                            listenersArray=null;
130                    }
131            }
132    
133            public void removeListener(PenManagerListener l) {
134                    synchronized(listeners) {
135                            listeners.remove(l);
136                            listenersArray=null;
137                    }
138            }
139    
140            PenManagerListener[] getListenersArray(){
141                    synchronized(listeners){
142                            if(listenersArray==null)
143                                    listenersArray=listeners.toArray(new PenManagerListener[listeners.size()]);
144                            return listenersArray;
145                    }
146            }
147    
148            public void firePenDeviceAdded(PenProvider.Constructor constructor, PenDevice device) {
149                    byte nextDeviceId=getNextDeviceId();
150                    device.penManagerSetId(nextDeviceId);
151                    if(deviceIdToDevice.put(nextDeviceId, device)!=null)
152                            throw new AssertionError();
153                    for(PenManagerListener l: getListenersArray()){
154                            l.penDeviceAdded(constructor, device);
155                    }
156            }
157    
158            private byte getNextDeviceId(){
159                    Set<Byte> deviceIds=deviceIdToDevice.keySet();
160                    while(deviceIds.contains(Byte.valueOf(nextDeviceId)))
161                            nextDeviceId++;
162                    if(nextDeviceId<0)
163                            throw new IllegalStateException();
164                    return nextDeviceId;
165            }
166    
167            public void firePenDeviceRemoved(PenProvider.Constructor constructor, PenDevice device) {
168                    if(deviceIdToDevice.remove(device.getId())==null)
169                            throw new IllegalArgumentException("device not found");
170                    for(PenManagerListener l: getListenersArray())
171                            l.penDeviceRemoved(constructor, device);
172            }
173    
174            public PenDevice getDevice(byte deviceId){
175                    return deviceIdToDevice.get(Byte.valueOf(deviceId));
176            }
177    
178            public Collection<PenDevice> getDevices(){
179                    return devicesA;
180            }
181    
182            public Set<PenProvider.Constructor> getConstructors() {
183                    return providerConstructorsA;
184            }
185    
186            void setPaused(boolean paused) {
187                    if(this.paused==paused)
188                            return;
189                    pen.scheduler.setPaused(paused);
190                    this.paused=paused;
191                    for(PenProvider.Constructor providerConstructor: providerConstructors){
192                            PenProvider penProvider=providerConstructor.getConstructed();
193                            if(penProvider!=null)
194                                    penProvider.penManagerPaused(paused);
195                    }
196            }
197    
198            public boolean getPaused() {
199                    return paused;
200            }
201    
202            /**
203            Schedules button events. You must construct a new {@code PButton} each time you call this method (do not reuse).
204            */
205            public void scheduleButtonEvent(PenDevice device, long deviceTime, PButton button) {
206                    if(paused)
207                            return;
208                    pen.scheduler.scheduleButtonEvent(device, deviceTime, button);
209            }
210    
211            /**
212            Schedules scroll events. You must construct a new {@code PScroll} each time you call this method (do not reuse).
213            */
214            public void scheduleScrollEvent(PenDevice device, long deviceTime, PScroll scroll) {
215                    if(paused)
216                            return;
217                    pen.scheduler.scheduleScrollEvent(device, deviceTime, scroll);
218            }
219            
220            public boolean scheduleLevelEvent(PenDevice device, long deviceTime, Collection<PLevel> levels) {
221                    return scheduleLevelEvent(device, deviceTime, levels, false);
222            }
223    
224            /**
225            Schedules level events. You can reuse the levels {@code Collection} but you must construct new {@code PLevel}s each time you call this method.
226            */
227            public boolean scheduleLevelEvent(PenDevice device, long deviceTime, Collection<PLevel> levels, boolean levelsOnScreen) {
228                    if(paused)
229                            return false;
230                    return pen.scheduler.scheduleLevelEvent(device, deviceTime, levels, levelsOnScreen);
231            }
