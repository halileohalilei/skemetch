019    package jpen;
020    
021    import java.util.Arrays;
022    import java.util.Collections;
023    import java.util.List;
024    
025    public class PButton
026                            extends TypedValuedClass<PButton.Type, Boolean>
027            implements java.io.Serializable {
028            public static final long serialVersionUID=1l;
029    
030            public enum Type{
031                    LEFT, CENTER, RIGHT,
032                    /**
033                    Used whenever the {@link Pen} {@link PLevel.Type#PRESSURE} level value changes from/to 0.
034                    */
035                    ON_PRESSURE, 
036                    CUSTOM;
037                    public static final List<Type> ALL_VALUES=Collections.unmodifiableList(Arrays.asList(values()));
038                    public static final List<Type> VALUES=TypedClass.createStandardTypes(ALL_VALUES);
039            }
040    
041            public PButton(int typeNumber, Boolean value) {
042                    super(typeNumber, value);
043            }
044            
045            // TODO: valueOf ?
046    
047            @Override
048            final List<Type> getAllTypes(){
049                    return Type.ALL_VALUES;
050            }
051    }
